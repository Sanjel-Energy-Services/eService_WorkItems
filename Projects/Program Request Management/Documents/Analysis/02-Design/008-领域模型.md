	+retrieveHistory()

	<<entity>>

class ApproverType {
	<<enumeration>>
	PEER_REVIEW
	MANAGER_REVIEW
	MANAGER_APPROVAL
	DIRECTOR_APPROVAL

	<<enumeration>>
	EMAIL_FORWARDING
	WEB_PORTAL_FORM

	<<enumeration>>

	<<enumeration>>

	MEDIUM  // Due within 2 days
	LOW     // > 2 days
	NORMAL  // No due date

	<<enumeration>>

}

	<<enumeration>>

	<<enumeration>>
	PENDING

	DELIVERED
	READ
	FAILED
}
